Sub 入庫チェック実行(ws As Worksheet)

    Dim lastRow As Long
    Dim dict As Object
    Dim i As Long
    Dim key As String
    
    lastRow = ws.Cells(ws.Rows.Count, "A").End(xlUp).Row

    If lastRow < 2 Then Exit Sub
    
    Set dict = CreateObject("Scripting.Dictionary")
    
    '=========================
    ' ① シート内重複チェック
    '=========================
    For i = 2 To lastRow
        
        key = Trim(ws.Cells(i, "F").value)
        ws.Cells(i, "H").value = "OK"
              
        If Len(ws.Cells(i, "F").value) <= 12 Then
            ws.Cells(i, "K").value = "NG(品番短い)"
            ws.Cells(i, "H").value = "error"
        ElseIf dict.Exists(key) Then
            ws.Cells(i, "K").value = "ダブりありNG"
            ws.Cells(i, "H").value = "error"
        Else
            dict.Add key, True
            ws.Cells(i, "K").value = "ダブりなしOK"
        End If
        
    Next i
    
    
    '=========================
    ' ② Access側チェック
    '=========================
    
    ロケーションチェック
    
End Sub
Sub ロケーションチェック()

    Dim db As New CDatabaseConnection
    Dim pathMager As New clsPathManager
    
    db.OpenDB pathMager.accdbPath

    Dim repo As New clsLocationRepository
    repo.SetConnection db

    Dim service As New clsLocationService
    service.Initialize repo

    service.LocationDuplication Sheets("free")

    db.CloseDB

End Sub

    service.TransferNyuko Sheets("free")

    service.Terminate

    MsgBox "処理完了"

End Sub

    If Len(C_t) <> 10 Then
        ws.Cells(i, 9).value = "NG"
        ws.Cells(i, 11).value = "error"
    End If
    
    ws.Cells(i, 12).value = C_t & ws.Cells(i, 7).value & ws.Cells(i, 8).value

Next i

Function isExternalProduct(ByVal targetValue As String) As Boolean
    
    Static dict As Object
    Dim wsList As Worksheet
    Dim lastRow As Long
    Dim arr
    Dim i As Long
    
    ' 初回だけDictionary作成
    If dict Is Nothing Then
        
        Set dict = CreateObject("Scripting.Dictionary")
        Set wsList = ThisWorkbook.Worksheets("10桁品番")
        
        lastRow = wsList.Cells(wsList.Rows.Count, 1).End(xlUp).Row
        arr = wsList.Range("A1:A" & lastRow).value
        
        For i = 1 To UBound(arr, 1)
            If arr(i, 1) <> "" Then
                dict(arr(i, 1)) = True
            End If
        Next
        
    End If
    
    ' 存在しないなら True
    isExternalProduct = Not dict.Exists(targetValue)
    
End Function

Public Sub Execute()

    Dim ws As Worksheet
    Dim filePath As String, db_Path As String
    Dim m_path As New clsPathManager
    
'    On Error GoTo ErrHandler
    
    Set ws = ThisWorkbook.Worksheets("free")

Cleanup:
    m_db.CloseDB
    Exit Sub

ErrHandler:
    MsgBox "CSV取込エラー：" & Err.Description, vbExclamation
    Resume Cleanup

End Sub

Private Sub ReadCsv(ByVal filePath As String, ByVal ws As Worksheet)

    Dim stm As ADODB.Stream
    Dim lines As Variant
    Dim cols As Variant
    Dim i As Long, j As Long

    ws.Cells.Clear

    Set stm = New ADODB.Stream

    With stm
        .Charset = "UTF-8"
        .Open
        .LoadFromFile filePath

        lines = Split(.ReadText, vbCrLf)

        For i = 0 To UBound(lines)

            If Trim(lines(i)) <> "" Then

                cols = Split(lines(i), ",")

                For j = 0 To UBound(cols)
                    ws.Cells(i + 2, j + 1).value = cols(j)
                Next j

            End If

        Next i

        .Close
    End With

    Set stm = Nothing

End Sub

'-----------------------------
' ロケ重複チェック
'-----------------------------
Public Sub LocationDuplication(ByVal ws As Worksheet)

    Dim i As Long, lastRow As Long
    Dim location As String, itemID As String
    Dim firstChar As String

    lastRow = ws.Cells(ws.Rows.Count, 5).End(xlUp).Row

    For i = 2 To lastRow

        location = Trim(ws.Cells(i, "E").value)
        itemID = Trim(ws.Cells(i, "F").value)

        '------------------------
        ' ロケーションチェック
        '------------------------
        If Len(location) > 0 Then

            firstChar = UCase$(Left$(location, 1))

            Select Case firstChar
                Case "1", "2", "3", "4", _
                     "B", "C", "D", "E", _
                     "X", "Y", "Z"

                    If m_repo.ExistsShukkaLocation(location) Then
                        ws.Cells(i, "I").value = "満室NG"
                        ws.Cells(i, "H").value = "error"
                    Else
                        ws.Cells(i, "I").value = "空室"
                    End If

                Case Else
                    ws.Cells(i, "I").value = "FREE"
            End Select

        End If

        '------------------------
        ' 品番チェック
        '------------------------
        If itemID <> "" Then

            If Not m_repo.ExistsItemID(itemID) Then
                ws.Cells(i, "J").value = "品番なし"
                ws.Cells(i, "K").value = "error"
            Else
                ws.Cells(i, "J").value = "品番OK"
            End If

        End If

    Next i

End Sub

'-----------------------------
' 入庫転送（トランザクション）
'-----------------------------
Public Sub TransferNyuko(ByVal ws As Worksheet)

    Dim i As Long, lastRow As Long
    Dim dbTran As New CDatabaseTransaction

    On Error GoTo ErrHandler

    lastRow = ws.Cells(ws.Rows.Count, 1).End(xlUp).Row

    dbTran.Initialize m_db.Connection
    dbTran.BeginTrans

    For i = 2 To lastRow

        If ws.Cells(i, 8).value <> "error" Then

            m_repo.UpdateNyuko _
                ws.Cells(i, 6).value, _
                ws.Cells(i, 5).value, _
                ws.Cells(i, 1).value, _
                ws.Cells(i, 2).value, _
                ws.Cells(i, 7).value

        End If

    Next i

    dbTran.Commit
    Exit Sub

ErrHandler:
    dbTran.Rollback
    MsgBox "入庫転送中にエラー：" & Err.Description

End Sub

'CDatabaseConnection
Private mConn As ADODB.Connection
Private Const MAX_RETRY As Long = 3
Private Const PROVIDER_NAME As String = "Microsoft.ACE.OLEDB.12.0"

Public Function TryOpenConnection(ByVal dbPath As String, _
                     Optional ByVal showMessage As Boolean = False) As Boolean

    Dim retry As Long
    
    For retry = 1 To MAX_RETRY
    
        On Error Resume Next
        
        If mConn Is Nothing Then
            Set mConn = New ADODB.Connection
            mConn.Provider = PROVIDER_NAME
        End If
        
        If mConn.State <> adStateOpen Then
            mConn.Open dbPath
        End If
        
        If Err.Number = 0 And mConn.State = adStateOpen Then
            TryOpenConnection = True
            On Error GoTo 0
            Exit Function
        End If
        
        Err.Clear
        On Error GoTo 0
        DoEvents
        
    Next retry
    
    If showMessage Then
        MsgBox "DB connection failed.", vbCritical
    End If
    
End Function
Public Sub OpenDB(ByVal dbPath As String)

    Set mConn = New ADODB.Connection
    mConn.Provider = PROVIDER_NAME
    mConn.Open dbPath

End Sub

Public Property Get Connection() As ADODB.Connection
    Set Connection = mConn
End Property

Public Sub CloseDB()

    If Not mConn Is Nothing Then
        If mConn.State = adStateOpen Then mConn.Close
    End If
    
    Set mConn = Nothing

End Sub
