Option Explicit
'AutoRunM
Public nextRunTime As Date
Public isRunning As Boolean

Sub StartAutoRun()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Worksheets("cover")
    With Sheets("cover")

        If isRunning Then Exit Sub

        isRunning = True
'        .Range("K1").value = 2
        .Range("L1").value = "自動更新起動中..."

    End With
    
    Call MoveShapeTopRight(ws.Shapes("WA380DOWN"), ws.Range("A12"))
    Call MoveShapeTopRight(ws.Shapes("pc200"), ws.Range("AA10"))
    Call MoveShapeTopRight(ws.Shapes("WA380UP"), ws.Range("AA10"))

    ScheduleNextRun

End Sub


Sub StopAutoRun()
    Dim ws As Worksheet
    Set ws = ThisWorkbook.Worksheets("cover")
    On Error Resume Next

    Application.OnTime nextRunTime, "AutoRunMain", Schedule:=False

    On Error GoTo 0

    isRunning = False

    With Sheets("cover")
'        .Range("K1").value = 1
        .Range("L1").value = "自動更新停止"
    End With
    
    Call MoveShapeTopRight(ws.Shapes("pc200"), ws.Range("C10"))
    Call MoveShapeTopRight(ws.Shapes("WA380DOWN"), ws.Range("AA10"))
    Call MoveShapeTopRight(ws.Shapes("WA380UP"), ws.Range("AA10"))
    
End Sub


Sub ScheduleNextRun()

    If isRunning = False Then Exit Sub

    nextRunTime = Now + TimeSerial(0, 0, 30)

    Application.OnTime nextRunTime, "AutoRunMain"

End Sub


Sub AutoRunMain()

    If isRunning = False Then Exit Sub

    '=========================
    ' メイン処理
    '=========================
    RunShhippingRequest
    '=========================

    ScheduleNextRun

End Sub
Sub MoveShapeTopRight(shp As Shape, rng As Range)

    shp.Top = rng.Top               '縦：セル上側に合わせる
    shp.Left = rng.Left '+ rng.Width - shp.Width   '横：セル右端 - 図の幅

End Sub

Public Function lastRow(ByVal ws As Worksheet, _
                              ByVal col As Long) As Long

    If Application.WorksheetFunction.CountA(ws.Columns(col)) = 0 Then
        lastRow = 1
    Else
        lastRow = ws.Cells(ws.Rows.count, col).End(xlUp).row
    End If

End Function
Public Function last_Row(ByVal ws As Worksheet, _
                              ByVal col As Long) As Long

    If Application.WorksheetFunction.CountA(ws.Columns(col)) = 0 Then
        last_Row = 1
    Else
        last_Row = ws.Cells(ws.Rows.count, col).End(xlUp).row
    End If

End Function

Function GetSelectedRowRange() As Variant

    My_Range.Select

    fstRow = Selection(1).row
    LstRow = Selection(Selection.count).row

    result(1) = fstRow
    result(2) = LstRow

    GetSelectedRowRange = result
    Exit Function

errhandler:
    GetSelectedRowRange = Array(0, 0)
End Function

Sub DeleteList(listType As listType)

    Dim db As New CDatabaseConnection
    Dim rs As New ADODB.Recordset
    Dim logger As New CDeleteLogger
    Dim pathM As New clsPathManager
    Dim ws As Worksheet
    Dim i As Long
    Dim tableName As String

    Set ws = ThisWorkbook.Worksheets("free")

    i = SelectRow
    If i = 0 Then Exit Sub
    
    If Not db.TryOpenConnection(pathM.accdbPath, True) Then Exit Sub
    
    logger.Initialize db

    Select Case listType
    
        Case ltShipping
            tableName = "出荷"
            
        Case ltDelivery
            tableName = "ロケーション"
            
    End Select

    rs.Open tableName, db.Connection, adOpenKeyset, adLockOptimistic
    
    rs.Find "ID=" & ws.Cells(i, "A").value

    If Not rs.EOF Then
        
        'ログ
        logger.LogDelete tableName, rs!Id, rs!品番
        
        '削除
        rs.Delete
        MsgBox "削除されました。"
    End If
    
    rs.Close
    db.CloseDB

End Sub

Sub 納入チェック()
Dim i As Long, C_t As String, Item_kind As String
Dim ws As Worksheet

Set ws = ThisWorkbook.Worksheets("free")

For i = 2 To lastRow(ws, 1)

    ws.Cells(i, 8).value = Format(ws.Cells(i, 1).value, "yyyymmdd") & "00"
    C_t = Left(ws.Cells(i, 5).value, 10)
    ws.Cells(i, 5).value = C_t
    ws.Cells(i, 9).value = "OK"
        
    If Len(C_t) <> 10 Then
        ws.Cells(i, 9).value = "NG"
        ws.Cells(i, 10).value = "品番が変"
    End If
        
    If IsExternalProduct(C_t) Then
        ws.Cells(i, 5).value = ws.Cells(i, 5).value & "EZ"
    Else
        ws.Cells(i, 5).value = ws.Cells(i, 5).value & "SH"
    End If
    
    ws.Cells(i, 12).value = ws.Cells(i, 5).value & _
            ws.Cells(i, 8).value & "000000" & ws.Cells(i, 7).value
            
    If Not CheckInventory(Trim(ws.Cells(i, "L").value)) Then
        ws.Cells(i, 10).value = "品番ok"
    Else
        ws.Cells(i, 10).value = "品番登録済"
        ws.Cells(i, 9).value = "NG"
    End If
    
Next i

End Sub

Function IsExternalProduct(ByVal targetValue As String) As Boolean
    
    Static dict As Object
    Dim wsList As Worksheet
    Dim lastRow As Long
    Dim arr
    Dim i As Long
    
    If dict Is Nothing Then
        
        Set dict = CreateObject("Scripting.Dictionary")
        dict.CompareMode = vbTextCompare   ' 大文字小文字無視
        
        Set wsList = ThisWorkbook.Worksheets("10桁品番")
        
        lastRow = wsList.Cells(wsList.Rows.count, 1).End(xlUp).row
        arr = wsList.Range("A1:A" & lastRow).value
        
        For i = 1 To UBound(arr, 1)
            If arr(i, 1) <> "" Then
                dict(CStr(arr(i, 1))) = True
            End If
        Next
        
    End If
    
    IsExternalProduct = Not dict.Exists(CStr(targetValue))
    
End Function

Sub 入庫チェック実行(ws As Worksheet)

    Dim lastRow As Long
    Dim dict As Object
    Dim i As Long
    Dim key As String
    
    lastRow = ws.Cells(ws.Rows.count, "A").End(xlUp).row

    If lastRow < 2 Then Exit Sub
    
    Set dict = CreateObject("Scripting.Dictionary")
    
    '=========================
    '  シート内重複チェック
    '=========================
    For i = 2 To lastRow
        
        key = Trim(ws.Cells(i, "F").value)
        ws.Cells(i, "H").value = "OK"
              
        If Len(ws.Cells(i, "F").value) <= 12 Then
            ws.Cells(i, "K").value = "NG(品番短い)"
            ws.Cells(i, "H").value = "error"
        ElseIf dict.Exists(key) Then
            ws.Cells(i, "K").value = "ダブりありNG"
            ws.Cells(i, "H").value = "error"
        Else
            dict.Add key, True
            ws.Cells(i, "K").value = "ダブりなしOK"
        End If
        
    Next i
        
    CheckLocation
    
End Sub

Sub FormatShippingList()
    Dim ws As Worksheet
    Dim i As Long, lastRow As Long
    Dim arrD
    Dim arrE()
    Dim arrH()
    
    Set ws = ThisWorkbook.Worksheets("free")
    
    ws.Columns("A:A").NumberFormatLocal = "m/d"
    lastRow = F_lastRow(ws, 4)
    With ws
    
        '高速化
        Application.ScreenUpdating = False
        
        '--- D列取得 ---
        arrD = .Range("D2:D" & lastRow).value
        
        ReDim arrE(1 To UBound(arrD), 1 To 1)
        ReDim arrH(1 To UBound(arrD), 1 To 1)

        '--- 文字抽出 ---
        For i = 1 To UBound(arrD)
        
            arrE(i, 1) = Mid(arrD(i, 1), 1, 12)
            arrH(i, 1) = Mid(arrD(i, 1), 29)
            
        Next i

        '--- 一括書き込み ---
        .Range("E2:E" & lastRow).value = arrE
        .Range("H2:H" & lastRow).value = arrH

        '--- XLOOKUP ---
        .Range("C2:C" & lastRow).Formula = _
        "=XLOOKUP(D2,出荷依頼!I:I,出荷依頼!B:B,"""",FALSE)"

        .Range("A2:A" & lastRow).Formula = _
        "=XLOOKUP(D2,出荷依頼!I:I,出荷依頼!A:A,"""",FALSE)"

        '--- 値変換 ---
        .Range("C2:C" & lastRow).value = .Range("C2:C" & lastRow).value
        .Range("A2:A" & lastRow).value = .Range("A2:A" & lastRow).value

        Application.ScreenUpdating = True

        End With

Function load_shipping_requestData() As Boolean
    Dim wb As Workbook, ws As Worksheet, wsSrc As Worksheet
    Dim filePath As String, fileTime As Date
    Dim Path_M As New clsPathManager
    Dim File_C As New clsFileChecker
    
    Set ws = ThisWorkbook.Worksheets("出荷")
    Set wsSrc = ThisWorkbook.Worksheets("cover")
    filePath = Path_M.xlsxPath
    
    If Not File_C.TryCheck(filePath, True) Then Exit Function
    
 'ファイル更新時間取得
    fileTime = FileDateTime(filePath)
    

    If wsSrc.Range("K1").value <> "" Then
        If DateDiff("n", wsSrc.Range("K1").value, fileTime) = 0 Then
                load_shipping_requestData = False
                Exit Function
        End If
    End If
    wsSrc.Range("K1").value = fileTime
    
    Application.ScreenUpdating = False

    ws.Cells.Clear

    Set wb = Workbooks.Open(fileName:=filePath, ReadOnly:=True, UpdateLinks:=0)

    ws.Range("A2").Resize(19, 12).value = _
        wb.Worksheets("出荷貼り付け").Range("A2:M20").value

    ws.Columns("A").NumberFormat = "yyyy/mm/dd"

    Application.DisplayAlerts = False
    wb.Close False
    Application.DisplayAlerts = True

    Application.ScreenUpdating = True
    DoEvents
    ActiveWindow.Zoom = ActiveWindow.Zoom
    
    load_shipping_requestData = True
    
End Function

Private Sub ReadCsv(ByVal filePath As String, ByVal ws As Worksheet)

    Dim stm As ADODB.Stream
    Dim lines As Variant
    Dim cols As Variant
    Dim i As Long, j As Long

    ws.Cells.Clear

    Set stm = New ADODB.Stream

    With stm
        .Charset = "UTF-8"
        .Open
        .LoadFromFile filePath

        lines = Split(.ReadText, vbCrLf)

        For i = 0 To UBound(lines)

            If Trim(lines(i)) <> "" Then

                cols = Split(lines(i), ",")

                For j = 0 To UBound(cols)
                    ws.Cells(i + 2, j + 1).value = cols(j)
                Next j

            End If

        Next i

        .Close
    End With

    Set stm = Nothing

End Sub

'CDatabaseConnection
Private mConn As ADODB.Connection
Private Const MAX_RETRY As Long = 3
Private Const PROVIDER_NAME As String = "Microsoft.ACE.OLEDB.12.0"

Public Function TryOpenConnection(ByVal dbPath As String, _
                     Optional ByVal showMessage As Boolean = False) As Boolean

    Dim retry As Long
    
    For retry = 1 To MAX_RETRY
    
        On Error Resume Next
        
        If mConn Is Nothing Then
            Set mConn = New ADODB.Connection
            mConn.Provider = PROVIDER_NAME
        End If
        
        If mConn.State <> adStateOpen Then
            mConn.Open dbPath
        End If
        
        If Err.Number = 0 And mConn.State = adStateOpen Then
            TryOpenConnection = True
            On Error GoTo 0
            Exit Function
        End If
        
        Err.Clear
        On Error GoTo 0
        DoEvents
        
    Next retry
    
    If showMessage Then
        MsgBox "DB connection failed after 3 retries.", vbCritical
    End If
    
End Function

Public Property Get Connection() As ADODB.Connection
    Set Connection = mConn
End Property

Public Sub CloseDB()

    If Not mConn Is Nothing Then
        If mConn.State = adStateOpen Then mConn.Close
    End If
    
    Set mConn = Nothing

End Sub
Public Sub OpenDB(ByVal dbPath As String)

    Set mConn = New ADODB.Connection
    mConn.Provider = PROVIDER_NAME
    mConn.Open dbPath

End Sub

        ByVal tableName As String, _
        ByVal fieldName As String, _
        ByVal fieldValue As Variant, _
        Optional ByVal whereSql As String = "") As Boolean

    Dim cmd As ADODB.Command
    Dim rs As ADODB.Recordset
    
    Set cmd = New ADODB.Command
    
    With cmd
    
        .ActiveConnection = mConn
        .CommandType = adCmdText
        
        .CommandText = _
            "SELECT TOP 1 1 FROM [" & tableName & "] " & _
            "WHERE [" & fieldName & "] = ? " & whereSql
        
        .Parameters.Append .CreateParameter("p1", adVariant, adParamInput, , fieldValue)
    
    End With
    
    Set rs = cmd.Execute
    
    ExistsByField = Not rs.EOF
    
    rs.Close
    
    Set rs = Nothing
    Set cmd = Nothing

End Function

'========================
Public Function TryCheck(ByVal filePath As String, _
                         Optional showMessage As Boolean = False) As Boolean

    Dim retry As Long

    For retry = 1 To MAX_RETRY

        On Error Resume Next

        If Dir(filePath) <> "" Then
            TryCheck = True
            On Error GoTo 0
            Exit Function
        End If

        Err.Clear
        On Error GoTo 0
        DoEvents

    Next retry

    If showMessage Then
        MsgBox "ファイルが見つかりません：" & vbCrLf & filePath, vbCritical
    End If

End Function

Public Sub LocationDuplication(ByVal ws As Worksheet)

    Dim i As Long, lastRow As Long
    Dim Location As String, ItemID As String
    Dim firstChar As String

    lastRow = ws.Cells(ws.Rows.count, 5).End(xlUp).row

    For i = 2 To lastRow

        Location = Trim(ws.Cells(i, "E").value)
        ItemID = Trim(ws.Cells(i, "F").value)

        '------------------------
        ' ロケーションチェック
        '------------------------
        If Len(Location) > 0 Then

            firstChar = UCase$(Left$(Location, 1))

            Select Case firstChar
                Case "1", "2", "3", "4", _
                     "B", "C", "D", "E", _
                     "X", "Y", "Z"

                    If m_repo.ExistsShukkaLocation(Location) Then
                        ws.Cells(i, "I").value = "満室NG"
                        ws.Cells(i, "H").value = "error"
                    Else
                        ws.Cells(i, "I").value = "空室"
                    End If

                Case Else
                    ws.Cells(i, "I").value = "FREE"
            End Select

        End If

        '------------------------
        ' 品番チェック
        '------------------------
        If ItemID <> "" Then

            If Not m_repo.ExistsItemID(ItemID) Then
                ws.Cells(i, "J").value = "品番なし"
                ws.Cells(i, "K").value = "error"
            Else
                ws.Cells(i, "J").value = "品番OK"
            End If

        End If

    Next i

End Sub

    Dim rs As ADODB.Recordset
    
    For i = LBound(prefixList) To UBound(prefixList)
    
        If prefixList(i) <> "" Then
        
            If inSql <> "" Then inSql = inSql & ","
            inSql = inSql & "'" & prefixList(i) & "'"
            
        End If
        
    Next i
    
    If inSql = "" Then Exit Function

    For r = 0 To UBound(data, 2)

            result(3, idx) = data(3, r) '納入日
            result(4, idx) = data(4, r) 'シリアル
            
        End If
        
    Next
    
    If idx = -1 Then Exit Function
    
    ReDim Preserve result(0 To 4, 0 To idx)
    
    GetShippingLis13t = result

Function TransposeArray(arr As Variant) As Variant

    Dim i As Long, j As Long
    Dim tmp()
    
    ReDim tmp(UBound(arr, 2), UBound(arr, 1))
    
    For i = 0 To UBound(arr, 1)
        For j = 0 To UBound(arr, 2)
            tmp(j, i) = arr(i, j)
        Next j
    Next i
    
    TransposeArray = tmp

End Function

        Exit Sub
    End If
